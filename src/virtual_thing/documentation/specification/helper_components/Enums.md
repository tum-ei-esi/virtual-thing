# RuntimeEvent
Type: `string`  
Allowed values:
- `"invokeAction"`
- `"subscribeEvent"`
- `"unsubscribeEvent"`
- `"readProperty"`
- `"writeProperty"`
- `"emitEvent"`
- `"startup"`
- `"shutdown"`

# ReadOperation
Type: `string`  
Allowed values:
- `"get"`
- `"pop"`
- `"copy"`
- `"length"`
- `"parse"`

# WriteOperation
Type: `string`  
Allowed values:
- `"set"`
- `"push"`
- `"copy"`
- `"pushCopy"`
- `"concat"`

# StateMachineOperation
Type: `string`  
Allowed values:
- `"transit"`
- `"reset"`
- `"error"`
